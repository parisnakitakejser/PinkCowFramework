Database

connect()
Description: connect to your database provider
@return void
Sample how to connect to database 
PinkCow::import('Database');
PinkCow\Database::setDBHost('{database-host}');
PinkCow\Database::setDBUser('{database-user}');
PinkCow\Database::setDBPass('{database-pass}');
PinkCow\Database::setDBName('{database-name}');
PinkCow\Database::connect();

fetch(query, bindparams)
Description: select a single row
@return object
query
Type: string
bindparams
Type: array
Defualt: array()
Sample: definde database host 
$feild2 = 'sample';
$query = PinkCow\Database::prepare("SELECT t1.feild1 FROM table1 t1 WHERE td.feild2 = :feild2 LIMIT 1");
$bindparam = array(
	array('feild2', $feild2, 'str')
);

$row = PinkCow\Database::fetch($query, $bindparam);
exec(query, bindparams)
Description: execute sql command like update or remove, not a select.
@return void
query
Type: string
bindparams
Type: array
Defualt: array()
Sample: definde database host 
$feild1 = 1337;
$feild2 = 'sample';
$query = PinkCow\Database::prepare("UPDATE table1 SET feild1 = :feild1 WHERE td.feild2 = :feild2 LIMIT 1");
$bindparam = array(
	array('feild1', $feild1, 'int'),
	array('feild2', $feild2, 'str'),
);

PinkCow\Database::exec($query, $bindparam);

@return int
